class Solution {
    public int lengthOfLongestSubstring(String sa) {
        int start = 0, pointer = 0, max = 0;
        Set<Character> use = new HashSet<>();
        int l = sa.length();
        char[] arr = sa.toCharArray();
        
        for (;;) {
            if (l == pointer) break;
            if (use.contains(arr[pointer])) {
                ++start;
                pointer = start;
                use.clear();
                continue;
            }
            use.add(arr[pointer]);
            max = Math.max(max, use.size());

            ++pointer;
        }
        
        return max;
    }
}
